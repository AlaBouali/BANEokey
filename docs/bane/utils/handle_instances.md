<pre><code class="python">import gc

def get_current_instances(instance_type):
    active = []
    inactive = []
    b = list(gc.get_objects())
    for x in b:
        try:
            if &#34;bane.&#34; + instance_type in x.__repr__():
                try:
                    if x.done() == True:
                        inactive.append(x)
                    else:
                        active.append(x)
                except:
                    pass
        except:
            pass
    return {&#34;active&#34;: active, &#34;inactive&#34;: inactive}</code></pre>
